```dart
import 'package:drift/drift.dart';
import 'package:drift_flutter/drift_flutter.dart';

part 'cache_database.g.dart';

/// 오프라인 캐시 콘텐츠.
class CachedContents extends Table {
  TextColumn get id => text()();
  TextColumn get markdown => text()();

  @override
  Set<Column> get primaryKey => {id};
}

@DriftDatabase(tables: [CachedContents])
class AppDatabase extends _$AppDatabase {
  AppDatabase(super.e);

  @override
  int get schemaVersion => 1;

  Future<void> upsertContent(String id, String markdown) =>
      into(cachedContents).insertOnConflictUpdate(
          CachedContentsCompanion.insert(id: id, markdown: markdown));

  Future<List<CachedContent>> allContents() => select(cachedContents).get();

  Stream<List<CachedContent>> watchContents() => select(cachedContents).watch();
}

/// 앱 실행용 — 네이티브 파일 DB(테스트는 NativeDatabase.memory 주입).
AppDatabase openAppDatabase() => AppDatabase(driftDatabase(name: 'devpath_cache'));
```

```dart
  const MobileShell({super.key, required this.navigationShell});

    (icon: DpIcons.home, label: '홈'),
    (icon: DpIcons.path, label: '경로'),
    (icon: DpIcons.mentor, label: '멘토'),
    (icon: DpIcons.community, label: '커뮤니티'),
    (icon: DpIcons.notifications, label: '알림'),
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: navigationShell,
      bottomNavigationBar: NavigationBar(
        selectedIndex: navigationShell.currentIndex,
        onDestinationSelected: (i) => navigationShell.goBranch(
          i,
          initialLocation: i == navigationShell.currentIndex,
        ),
        destinations: [
          for (final t in _tabs)
            NavigationDestination(icon: Icon(t.icon), label: t.label),
        ],
      ),
    );

}
```

> - iOS `Info.plist`에 `CFBundleURLTypes` → `CFBundleURLSchemes: [devpath]`.
> 등록 후 `devpath://notifications`가 `/notifications`로 라우팅(go_router가 URI path 매핑). 코드는 위 라우트로 충분.
